// 获取爱心容器和爱心元素 const heartContainer = document.querySelector('.heart-container'); const heart = document.querySelector('.heart'); // 创建小星星并添加到爱心中 for (let i = 0; i < 100; i++) { const star = document.createElement('div'); star.classList.add('star'); // 随机位置 const x = Math.random() * heart.offsetWidth; const y = Math.random() * heart.offsetHeight; // 设置小星星的位置 star.style.left = `${x}px`; star.style.top = `${y}px`; // 将小星星添加到爱心中 heart.appendChild(star); } // 添加动画效果（这里使用CSS动画） heart.style.animation = 'beat 1s infinite alternate'; // CSS动画定义 const styleSheet = new CSSStyleSheet(); styleSheet.insertRule(` @keyframes beat { to { transform: scale(1.2); background-color: #ff5d5d; /* 可以改变颜色 */ } } `); document.adoptedStyleSheets = [styleSheet];
